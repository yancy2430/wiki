添加依赖

        <dependency>
            <groupId>com.bingo.upup</groupId>

            <version>最新版本</version>
            <scope>compile</scope>
        </dependency>

添加镜像打包插件

<build>
        <plugins>
            <plugin>
                <groupId>org.springframework.boot</groupId>
                <artifactId>spring-boot-maven-plugin</artifactId>
            </plugin>
            <plugin>
                <groupId>com.spotify</groupId>
                <artifactId>docker-maven-plugin</artifactId>
                <version>1.2.0</version>
                <executions>
                    <!--执行 mvn package 时 自动 执行 mvn docker:build-->
                    <execution>
                        <id>build-image</id>
                        <phase>package</phase>
                        <goals>
                            <goal>build</goal>
                            <!--<goal>push</goal>-->
                        </goals>
                    </execution>
                </executions>
                <configuration>
                    <serverId>docker-hub</serverId>
                    <!--镜像名称-->
                    <baseImage>adoptopenjdk/openjdk11</baseImage>
                    <imageName>registry-vpc.cn-shenzhen.aliyuncs.com/phanpy/${project.name}</imageName>
                    <image>${project.artifactId}</image>
                    <imageTags>
                        <imageTag>${project.version}</imageTag>
                        <imageTag>latest</imageTag>
                    </imageTags>
                    <env>
                        <TZ>Asia/Shanghai</TZ>
                        <ACTIVE>test</ACTIVE>
                    </env>
                    <cmd>java -jar -Dspring.profiles.active=$ACTIVE ${project.build.finalName}.jar</cmd>
                    <resources>
                        <resource>
                            <!-- 编译jar路径 -->
                            <!--<targetPath>/</targetPath>-->
                            <!--编译jar 所在文件夹-->
                            <directory>${project.build.directory}</directory>
                            <!--编译的jar 名称-->
                            <include>${project.name}-${project.version}.jar</include>
                        </resource>
                    </resources>
                </configuration>
            </plugin>
        </plugins>
    </build>

添加配置

在application.properties文件添加

place.airline=9c #航司二字码
spring.application.name=place-${place.airline} 应用名
spring.profiles.active=dev #开发环境 dev和test 可以不用管部署会自动更改
spring.sleuth.sampler.probability=1.0

回调结果参数说明


@Data
@Accessors(chain = true)
public class R<T> implements Serializable {
 /**
     * 结果code 0, "处理成功" 1, "可以重试" 2, "不可重试" 3, "换号重试"
     */
    private int code;
    /**
     * 结果集 航司返回原文 后台根据结果解析
     */
    private T data;
    /**
     * 错误描述
     */
    private String msg;

    /**
     * 请求参数原文返回
     */
    private Object param;
}

异常处理

下单过程异常中断统一抛出异常

//抛出异常 并传入错误原因和异常类型
throw new PlaceException("错误原因",ResultEnum.RETRY);

/**
 * 异常枚举类
 **/
public enum ResultEnum {
    SUCCEED(0, "处理成功"),
    RETRY(1, "可以重试"),
    NOTRETRY(2, "不可重试"),;
    private Integer code;
    private String desc;
    ResultEnum(int code, String desc) {
        this.code = code;
        this.desc = desc;
    }
}

接收结果回调

# 支付通知地址
http://120.78.91.205:8251/bingo_b2b_deal_with.php?action=set_chb2b_Pay&airline={航司二字码}

# 获取票号通知地址
http://120.78.91.205:8251/bingo_b2b_deal_with.php?action=set_chb2b_Ticket&airline={航司二字码}

通知地址返回 0 或 HTTP状态200为通知成功 否则失败
失败后会重复通知 第一次重复间隔 5秒 第二次 10秒 ... 通知5次 如一直不成功则去MQ查看死信队列
